import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import qmc

# Function
def f(x):
    return np.sin(x)

# Limits
a, b = 0, np.pi
N = 1000

# Sobol quasi-random samples
sampler = qmc.Sobol(d=1, scramble=True)
x_quasi = sampler.random_base2(m=10)

# Scale points
x_scaled = a + (b - a) * x_quasi.flatten()[:N]

# Function values
y = f(x_scaled)

# Monte Carlo Integration
integral_estimate = (b - a) * np.mean(y)
variance = np.var(y, ddof=1)

# DataFrame
df = pd.DataFrame({
    'x': x_scaled,
    'f(x)': y
})

# Output
print("OUTPUT\n")
print(df.head())

print("Monte Carlo Estimate of integral:",
      integral_estimate)

print("Variance of function values:",
      variance)

# Scatter Plot
plt.figure(figsize=(8,4))
sns.scatterplot(x=x_scaled, y=y, color='blue')

plt.title('Quasi-random Samples of sin(x)')
plt.xlabel('x')
plt.ylabel('f(x) = sin(x)')
plt.grid(True)

plt.show()

# Histogram
plt.figure(figsize=(8,4))

sns.histplot(y,
             bins=20,
             kde=True,
             color='green')

plt.title('Distribution of f(x) Values')
plt.xlabel('f(x)')
plt.ylabel('Frequency')

plt.show()
